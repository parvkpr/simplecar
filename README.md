# simplecar
A simple point car environment with openAI gym like interface
# usage
 from unicyclev2 import Env, run
 if __name__ == '__main__':
     e = Env()
     run(e)
     e.reset()
     for i in range(20):
         ac = e.action_space.sample()
         print(ac)
         print(e.step(ac))
         print('executing now' + repr(i))
